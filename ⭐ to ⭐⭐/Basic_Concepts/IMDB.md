# cook your dish here
for i in range(int(input())):
    n,x = map(int, input().split())
    
    arr = []
    for j in range(n):
        arr.append(list(map(int , input().split())))
        
    max_rate = 0
    for j in arr:
        if j[0] <= x:
            max_rate = max(max_rate,j[1])
    
    print(max_rate)
