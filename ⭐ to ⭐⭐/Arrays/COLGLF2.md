# cook your dish here
for i in range(int(input())):
    no_of_season=int(input())
    l1=list(map(int,input().split()))
    total_count=0
    for j in range(no_of_season):
        l2=list(map(int,input().split()))
        if len(l2)>=3:
           total_count+=sum([x-l1[j] for x in l2[2:]])+l2[1]
        else:
            total_count+=l2[1]
