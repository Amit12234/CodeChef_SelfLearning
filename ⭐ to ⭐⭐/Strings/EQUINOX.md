def check(s):
    if s[0] in "EQUINOX":
        return True
    return False
    
for i in range(int(input())):
    x,y,z = map(int, input().split())
    sarthak , anuradha = 0 , 0 
    for j in range(x):
        if check(input()):
            sarthak += y
        else:
            anuradha += z

    
    if(sarthak < anuradha):
        print("ANURADHA")
    elif(sarthak == anuradha):
        print("DRAW")
    else:
        print("SARTHAK")
